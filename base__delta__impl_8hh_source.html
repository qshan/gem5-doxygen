68 template <class BaseType, std::size_t DeltaSizeBits>
69 std::unique_ptr<BaseCacheCompressor::CompressionData>
70 BaseDelta<BaseType, DeltaSizeBits>::compress(const uint64_t* data,
71  Cycles& comp_lat, Cycles& decomp_lat)
72 {
73  std::unique_ptr<BaseCacheCompressor::CompressionData> comp_data =
74  DictionaryCompressor<BaseType>::compress(data);
75 
76  // If there are more bases than the maximum, the compressor failed.
77  // Otherwise, we have to take into account all bases that have not
78  // been used, considering that there is an implicit zero base that
79  // does not need to be added to the final size.
80  const int diff = DEFAULT_MAX_NUM_BASES -
81  DictionaryCompressor<BaseType>::numEntries;
82  if (diff < 0) {
83  comp_data->setSizeBits(DictionaryCompressor<BaseType>::blkSize * 8);
84  DPRINTF(CacheComp, "Base%dDelta%d compression failed\n",
85  8 * sizeof(BaseType), DeltaSizeBits);
86  } else if (diff > 0) {
87  comp_data->setSizeBits(comp_data->getSizeBits() +
88  8 * sizeof(BaseType) * diff);
89  }
90 
91  // Set compression latency (Assumes 1 cycle per entry and 1 cycle for
92  // packing)
93  comp_lat = Cycles(1 + (DictionaryCompressor<BaseType>::blkSize /
94  sizeof(BaseType)));
95 
96  // Set decompression latency
97  decomp_lat = Cycles(1);
98 
99  // Return compressed line
100  return comp_data;
101 }
