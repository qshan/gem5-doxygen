83 
84  int num_expr = tokens.size();
85  for (int i = 0; i < num_expr; ++i) {

87  int jstop = token.size();
88

91  if (j >= ntsize)

93 
94  const string &var = token[j];
95  if (var != "*" && var != name_tokens[j]) {
96  match = false;
97  break;

99  }
100 
101  if (match)
102  return true;
103  }
104 
105  return false;
