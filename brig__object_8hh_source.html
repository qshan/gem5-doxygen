121 // Utility function to bump Brig item pointer to next element given
122 // item size in bytes. Really just an add but with lots of casting.
123 template<typename T>
124 T*
125 brigNext(T *ptr)
126 {
127  Brig::BrigBase *base_ptr = (Brig::BrigBase*)ptr;
128  int size = base_ptr->byteCount;
129  assert(size);
130 
131  return (T*)((uint8_t*)ptr + size);
132 }
