131  findRegArrayMSB(ApicRegIndex base)
132  {
133  int offset = 7;
134  do {
135  if (regs[base + offset] != 0) {
136  return offset * 32 + findMsbSet(regs[base + offset]);
137  }
138  } while (offset--);
