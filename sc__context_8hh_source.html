176 template <class T>
177 inline void
178 sc_global<T>::update()
179 {
180  void *p = (::sc_gem5::Process *)sc_core::sc_get_current_process_handle();
181  if (p != m_proc) {
182  const T *vp = m_map[p];
183  if (vp == 0) {
184  vp = new T(sc_without_context());
185  m_map.emplace(p, vp);
186  }
187  m_proc = p;
188  m_value_ptr = vp;
189  }
190 }
