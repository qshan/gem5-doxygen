38 void
39 PowerStaticInst::printReg(std::ostream &os, RegId reg) const
40 {
41  if (reg.isIntReg())
42  ccprintf(os, "r%d", reg.index());
43  else if (reg.isFloatReg())
44  ccprintf(os, "f%d", reg.index());
45  else if (reg.isMiscReg())
46  switch (reg.index()) {
47  case 0: ccprintf(os, "cr"); break;
48  case 1: ccprintf(os, "xer"); break;
49  case 2: ccprintf(os, "lr"); break;
50  case 3: ccprintf(os, "ctr"); break;
51  default: ccprintf(os, "unknown_reg");
52  break;
53  }
54  else if (reg.isCCReg())
55  panic("printReg: POWER does not implement CCRegClass\n");
56 }
