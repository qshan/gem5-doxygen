54 SparcSystem::SparcSystem(Params *p)
55  : System(p), sysTick(0)
56 {
57  resetSymtab = new SymbolTable;
58  hypervisorSymtab = new SymbolTable;
59  openbootSymtab = new SymbolTable;
60  nvramSymtab = new SymbolTable;
61  hypervisorDescSymtab = new SymbolTable;
62  partitionDescSymtab = new SymbolTable;
63 
64  reset = loadFirmwareImage(params()->reset_bin, "reset binary");
65  openboot = loadFirmwareImage(params()->openboot_bin, "openboot binary");
66  hypervisor = loadFirmwareImage(
67  params()->hypervisor_bin, "hypervisor binary");
68  nvram = loadFirmwareImage(params()->nvram_bin, "nvram image");
69  hypervisor_desc = loadFirmwareImage(
70  params()->hypervisor_desc_bin, "hypervisor description image");
71  partition_desc = loadFirmwareImage(
72  params()->partition_desc_bin, "partition description image");
73 
74  // load symbols
75  if (!reset->loadGlobalSymbols(resetSymtab))
76  panic("could not load reset symbols\n");
77 
78  if (!openboot->loadGlobalSymbols(openbootSymtab))
79  panic("could not load openboot symbols\n");
80 
81  if (!hypervisor->loadLocalSymbols(hypervisorSymtab))
82  panic("could not load hypervisor symbols\n");
83 
84  if (!nvram->loadLocalSymbols(nvramSymtab))
85  panic("could not load nvram symbols\n");
86 
87  if (!hypervisor_desc->loadLocalSymbols(hypervisorDescSymtab))
88  panic("could not load hypervisor description symbols\n");
89 
90  if (!partition_desc->loadLocalSymbols(partitionDescSymtab))
91  panic("could not load partition description symbols\n");
92 
93  // load symbols into debug table
94  if (!reset->loadGlobalSymbols(debugSymbolTable))
95  panic("could not load reset symbols\n");
96 
97  if (!openboot->loadGlobalSymbols(debugSymbolTable))

106  panic("could not load hypervisor symbols\n");
107

122 void
123 writeFirmwareImage(ObjectFile *obj, Addr addr, const PortProxy &proxy)
124 {
125  MemoryImage image = obj->buildImage();
126 
127  // If the entry point isn't somewhere in the image, we assume we need to
128  // move where it's loaded so that it is.
129  if (addr < image.minAddr() || addr >= image.maxAddr()) {
130  // Move the image by the difference between the expected entry address,
131  // and the entry point in the object file.
132  image.offset(addr - obj->entryPoint());
133  }
134 
135  image.write(proxy);
136 }
