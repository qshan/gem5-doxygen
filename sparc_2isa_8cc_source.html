80 void
81 ISA::reloadRegMap()
82 {
83  installGlobals(gl, CurrentGlobalsOffset);
84  installWindow(cwp, CurrentWindowOffset);
85  // Microcode registers.
86  for (int i = 0; i < NumMicroIntRegs; i++)
87  intRegMap[MicroIntOffset + i] = i + TotalGlobals + NWindows * 16;
88  installGlobals(gl, NextGlobalsOffset);
89  installWindow(cwp - 1, NextWindowOffset);
90  installGlobals(gl, PreviousGlobalsOffset);
91  installWindow(cwp + 1, PreviousWindowOffset);
92 }

94 void
95 ISA::installWindow(int cwp, int offset)
96 {
97  assert(offset >= 0 && offset + NumWindowedRegs <= NumIntRegs);
98  RegIndex *mapChunk = intRegMap + offset;
99  for (int i = 0; i < NumWindowedRegs; i++)
100  mapChunk[i] = TotalGlobals +
101  ((i - cwp * RegsPerWindow + TotalWindowed) % (TotalWindowed));
102 }

104 void
105 ISA::installGlobals(int gl, int offset)
106 {
107  assert(offset >= 0 && offset + NumGlobalRegs <= NumIntRegs);
108  RegIndex *mapChunk = intRegMap + offset;
109  mapChunk[0] = 0;
110  for (int i = 1; i < NumGlobalRegs; i++)
111  mapChunk[i] = i + gl * NumGlobalRegs;
112 }

338 RegVal
339 ISA::readMiscReg(int miscReg, ThreadContext * tc)
340 {
341  switch (miscReg) {
342  // tick and stick are aliased to each other in niagra
343  // well store the tick data in stick and the interrupt bit in tick
344  case MISCREG_STICK:
345  case MISCREG_TICK:
346  case MISCREG_PRIVTICK:
347  // I'm not sure why legion ignores the lowest two bits, but we'll go
348  // with it
349  // change from curCycle() to instCount() until we're done with legion
350  DPRINTF(Timer, "Instruction Count when TICK read: %#X stick=%#X\n",
351  tc->getCpuPtr()->instCount(), stick);
352  return mbits(tc->getCpuPtr()->instCount() + (int64_t)stick,62,2) |
353  mbits(tick,63,63);
354  case MISCREG_FPRS:
355  // in legion if fp is enabled du and dl are set
356  return fprs | 0x3;
357  case MISCREG_PCR:
358  case MISCREG_PIC:
359  panic("Performance Instrumentation not impl\n");
360  case MISCREG_SOFTINT_CLR:
361  case MISCREG_SOFTINT_SET:
362  panic("Can read from softint clr/set\n");
363  case MISCREG_SOFTINT:
364  case MISCREG_TICK_CMPR:
365  case MISCREG_STICK_CMPR:
366  case MISCREG_HINTP:
367  case MISCREG_HTSTATE:
368  case MISCREG_HTBA:
369  case MISCREG_HVER:
370  case MISCREG_STRAND_STS_REG:
371  case MISCREG_HSTICK_CMPR:
372  case MISCREG_QUEUE_CPU_MONDO_HEAD:
373  case MISCREG_QUEUE_CPU_MONDO_TAIL:
374  case MISCREG_QUEUE_DEV_MONDO_HEAD:
375  case MISCREG_QUEUE_DEV_MONDO_TAIL:
376  case MISCREG_QUEUE_RES_ERROR_HEAD:
377  case MISCREG_QUEUE_RES_ERROR_TAIL:
378  case MISCREG_QUEUE_NRES_ERROR_HEAD:
379  case MISCREG_QUEUE_NRES_ERROR_TAIL:
380  case MISCREG_HPSTATE:
381  return readFSReg(miscReg, tc);
382  }
383  return readMiscRegNoEffect(miscReg);
384 }

567 void
568 ISA::setMiscReg(int miscReg, RegVal val, ThreadContext * tc)
569 {
570  RegVal new_val = val;
571 
572  switch (miscReg) {
573  case MISCREG_ASI:
574  tc->getDecoderPtr()->setContext(val);
575  break;
576  case MISCREG_STICK:
577  case MISCREG_TICK:
578  // stick and tick are same thing on niagra
579  // use stick for offset and tick for holding intrrupt bit
580  stick = mbits(val,62,0) - tc->getCpuPtr()->instCount();
581  tick = mbits(val,63,63);
582  DPRINTF(Timer, "Writing TICK=%#X\n", val);
583  break;
584  case MISCREG_FPRS:
585  // Configure the fpu based on the fprs
586  break;
587  case MISCREG_PCR:
588  // Set up performance counting based on pcr value
589  break;
590  case MISCREG_PSTATE:
591  pstate = val & PstateMask;
592  return;
593  case MISCREG_TL:
594  {
595  tl = val;
596  if (hpstate.tlz && tl == 0 && !hpstate.hpriv)
597  tc->getCpuPtr()->postInterrupt(0, IT_TRAP_LEVEL_ZERO, 0);
598  else
599  tc->getCpuPtr()->clearInterrupt(0, IT_TRAP_LEVEL_ZERO, 0);
600  return;
601  }
602  case MISCREG_CWP:
603  new_val = val >= NWindows ? NWindows - 1 : val;
604  if (val >= NWindows)
605  new_val = NWindows - 1;
606 
607  installWindow(new_val, CurrentWindowOffset);
608  installWindow(new_val - 1, NextWindowOffset);
609  installWindow(new_val + 1, PreviousWindowOffset);
610  break;
611  case MISCREG_GL:
612  installGlobals(val, CurrentGlobalsOffset);
613  installGlobals(val, NextGlobalsOffset);
614  installGlobals(val, PreviousGlobalsOffset);
615  break;
616  case MISCREG_PIL:
617  case MISCREG_SOFTINT:
618  case MISCREG_SOFTINT_SET:
619  case MISCREG_SOFTINT_CLR:
620  case MISCREG_TICK_CMPR:
621  case MISCREG_STICK_CMPR:
622  case MISCREG_HINTP:
623  case MISCREG_HTSTATE:
624  case MISCREG_HTBA:
625  case MISCREG_HVER:
626  case MISCREG_STRAND_STS_REG:
627  case MISCREG_HSTICK_CMPR:
628  case MISCREG_QUEUE_CPU_MONDO_HEAD:
629  case MISCREG_QUEUE_CPU_MONDO_TAIL:
630  case MISCREG_QUEUE_DEV_MONDO_HEAD:
631  case MISCREG_QUEUE_DEV_MONDO_TAIL:
632  case MISCREG_QUEUE_RES_ERROR_HEAD:
633  case MISCREG_QUEUE_RES_ERROR_TAIL:
634  case MISCREG_QUEUE_NRES_ERROR_HEAD:
635  case MISCREG_QUEUE_NRES_ERROR_TAIL:
636  case MISCREG_HPSTATE:
637  setFSReg(miscReg, val, tc);
638  return;
639  }
640  setMiscRegNoEffect(miscReg, new_val);
641 }
