358  size_t operator()(const X86ISA::ExtMachInst &emi) const {
359  return (((uint64_t)emi.legacy << 48) |

363  ((uint64_t)emi.sib << 16) |

365  ((uint64_t)emi.opcode.op)) ^
366  emi.immediate ^ emi.displacement ^
367  emi.mode ^
