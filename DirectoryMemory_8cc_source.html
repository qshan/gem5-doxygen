98 uint64_t
99 DirectoryMemory::mapAddressToLocalIdx(Addr address)
100 {
101  uint64_t ret = 0;
102  for (const auto& r: addrRanges) {
103  if (r.contains(address)) {
104  ret += r.getOffset(address);
105  break;
106  }
107  ret += r.size();
108  }
109  return ret >> RubySystem::getBlockSizeBits();
110 }
