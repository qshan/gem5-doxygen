58 static int
59 dumpDmesgEntry(const uint8_t *base, const uint8_t *end, const ByteOrder bo,
60  std::ostream &os)
61 {
62  const size_t max_length = end - base;
63  DmesgEntry de;
64 
65  if (max_length < sizeof(de)) {
66  warn("Malformed dmesg entry\n");
67  return -1;
68  }
69 
70  memcpy(&de, base, sizeof(de));
71  de.ts_nsec = gtoh(de.ts_nsec, bo);
72  de.len = gtoh(de.len, bo);
73  de.text_len = gtoh(de.text_len, bo);
74 
75  if (de.len < sizeof(de) ||
76  max_length < de.len ||
77  max_length < sizeof(DmesgEntry) + de.text_len) {
78 
79  warn("Malformed dmesg entry:\n");
80  warn("\tMax length: %i\n", max_length);
81  warn("\tde.len: %i\n", de.len);
82  warn("\tde.text_len: %i\n", de.text_len);
83  return -1;
84  }
85 
86  ccprintf(os, "[%.6f] ", de.ts_nsec * 10e-9);
87  os.write((char *)base + sizeof(de), de.text_len);
88  os << std::endl;
89 
90  return de.len;
91 }

93 void
94 Linux::dumpDmesg(ThreadContext *tc, std::ostream &os)
95 {
96  System *system = tc->getSystemPtr();
97  const ByteOrder bo = system->getGuestByteOrder();
98  const SymbolTable *symtab = system->kernelSymtab;
99  PortProxy &proxy = tc->getVirtProxy();
100 
101  Addr addr_lb = 0, addr_lb_len = 0, addr_first = 0, addr_next = 0;
102  const bool found_symbols =
103  symtab->findAddress("__log_buf", addr_lb) &&
104  symtab->findAddress("log_buf_len", addr_lb_len) &&
105  symtab->findAddress("log_first_idx", addr_first) &&
106  symtab->findAddress("log_next_idx", addr_next);
107 
108  if (!found_symbols) {
109  warn("Failed to find kernel dmesg symbols.\n");
110  return;
111  }
112 
113  uint32_t log_buf_len =
114  proxy.read<uint32_t>(addr_lb_len, TheISA::GuestByteOrder);
115  uint32_t log_first_idx =
116  proxy.read<uint32_t>(addr_first, TheISA::GuestByteOrder);
117  uint32_t log_next_idx =
118  proxy.read<uint32_t>(addr_next, TheISA::GuestByteOrder);
119 
120  if (log_first_idx >= log_buf_len || log_next_idx >= log_buf_len) {
121  warn("dmesg pointers/length corrupted\n");
122  return;
123  }
124 
125  // Normalize and read the dmesg ring buffer
126  std::vector<uint8_t> log_buf(log_buf_len);
127  int length;
128  if (log_first_idx < log_next_idx) {
129  length = log_next_idx - log_first_idx;
130  if (length < 0 || length > log_buf.size()) {
131  warn("Unexpected dmesg buffer length\n");
132  return;
133  }
134  proxy.readBlob(addr_lb + log_first_idx, log_buf.data(), length);
135  } else {
136  const int length_2 = log_buf_len - log_first_idx;
137  if (length_2 < 0 || length_2 + log_next_idx > log_buf.size()) {
138  warn("Unexpected dmesg buffer length\n");
139  return;
140  }
141  length = log_buf_len;
142  proxy.readBlob(addr_lb + log_first_idx, log_buf.data(), length_2);
143  proxy.readBlob(addr_lb, log_buf.data() + length_2, log_next_idx);
144  }
145 
146  // Print dmesg buffer content
147  const uint8_t *cur = log_buf.data(), *end = log_buf.data() + length;
148  while (cur < end) {
149  int ret = dumpDmesgEntry(cur, end, bo, os);
150  if (ret < 0)
151  return;
152  cur += ret;
153  }
154 }
