41 static void
42 handlePseudoInst(ThreadContext *xc, Packet *pkt)
43 {
44  const Addr offset(pkt->getAddr() & IPR_IN_CLASS_MASK);
45  const uint8_t func((offset >> 8) & 0xFF);
46  const uint8_t subfunc(offset & 0xFF);
47  uint64_t ret;
48 
49  assert((offset >> 16) == 0);
50  ret = PseudoInst::pseudoInst(xc, func, subfunc);
51  if (pkt->isRead())
52  pkt->set(ret, TheISA::GuestByteOrder);
53 }
