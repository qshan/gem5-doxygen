36 MemoryImage::writeSegment(const Segment &seg, const PortProxy &proxy) const
37 {
38  if (seg.size != 0) {
39  if (seg.data) {
40  proxy.writeBlob(seg.base, seg.data, seg.size);
41  } else {
42  // no image: must be bss

44  }
45  }
