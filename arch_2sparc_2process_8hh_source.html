83 
84  // Reserve 8M for main stack.
85  Addr max_stack_size = 8 * 1024 * 1024;
86 
87  // Set up stack. On SPARC Linux, stack goes from the top of memory
88  // downward, less the hole for the kernel address space.
89  Addr stack_base = 0xf0000000ULL;
90 
91  // Set pointer for next thread stack.
92  Addr next_thread_stack_base = stack_base - max_stack_size;
93 
94  // Set up region for mmaps.
