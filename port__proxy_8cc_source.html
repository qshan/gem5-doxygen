114 bool
115 PortProxy::tryReadString(char *str, Addr addr, size_t maxlen) const
116 {
117  assert(maxlen);
118  while (maxlen--) {
119  if (!tryReadBlob(addr++, str, 1))
120  return false;
121  if (!*str++)
122  return true;
123  }
124  // We ran out of room, so back up and add a terminator.
125  *--str = '\0';
