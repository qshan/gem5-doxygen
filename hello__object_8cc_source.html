58 void
59 HelloObject::processEvent()
60 {
61  timesLeft--;
62  DPRINTF(HelloExample, "Hello world! Processing the event! %d left\n",
63  timesLeft);
64 
65  if (timesLeft <= 0) {
66  DPRINTF(HelloExample, "Done firing!\n");
67  goodbye->sayGoodbye(myName);
68  } else {
69  schedule(event, curTick() + latency);
70  }
71 }
