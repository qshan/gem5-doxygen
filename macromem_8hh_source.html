51 static inline unsigned int
52 number_of_ones(int32_t val)
53 {
54  uint32_t ones = 0;
55  for (int i = 0; i < 32; i++ )
56  {
57  if ( val & (1<<i) )
58  ones++;
59  }
60  return ones;
61 }
