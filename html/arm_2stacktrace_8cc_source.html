48 readSymbol(ThreadContext *tc, const std::string name)
49 {

69 Addr
70 ProcessInfo::task(Addr ksp) const
71 {
72  Addr base = ksp & ~0x1fff;
73  if (base == ULL(0xffffffffc0000000))
74  return 0;
75 
76  Addr tsk;
77 
78  PortProxy &vp = tc->getVirtProxy();

100 ProcessInfo::name(Addr ksp) const
101 {

104  return "unknown";

106  char comm[256];
