332 template <class OS>
333 SyscallReturn
334 futexFunc(SyscallDesc *desc, int callnum, ThreadContext *tc)
335 {
336  using namespace std;
337 
338  int index = 0;
339  auto process = tc->getProcessPtr();
340 
341  Addr uaddr = process->getSyscallArg(tc, index);
342  int op = process->getSyscallArg(tc, index);
343  int val = process->getSyscallArg(tc, index);
344  int timeout M5_VAR_USED = process->getSyscallArg(tc, index);
345  Addr uaddr2 M5_VAR_USED = process->getSyscallArg(tc, index);
346  int val3 = process->getSyscallArg(tc, index);
347 
348  /*
349  * Unsupported option that does not affect the correctness of the
350  * application. This is a performance optimization utilized by Linux.
351  */
352  op &= ~OS::TGT_FUTEX_PRIVATE_FLAG;
353  op &= ~OS::TGT_FUTEX_CLOCK_REALTIME_FLAG;
354 
355  FutexMap &futex_map = tc->getSystemPtr()->futexMap;
356 
357  if (OS::TGT_FUTEX_WAIT == op || OS::TGT_FUTEX_WAIT_BITSET == op) {
358  // Ensure futex system call accessed atomically.
359  BufferArg buf(uaddr, sizeof(int));
360  buf.copyIn(tc->getVirtProxy());
361  int mem_val = *(int*)buf.bufferPtr();
362 
363  /*
364  * The value in memory at uaddr is not equal with the expected val
365  * (a different thread must have changed it before the system call was
366  * invoked). In this case, we need to throw an error.
367  */
368  if (val != mem_val)
369  return -OS::TGT_EWOULDBLOCK;
370 
371  if (OS::TGT_FUTEX_WAIT == op) {
372  futex_map.suspend(uaddr, process->tgid(), tc);
373  } else {
374  futex_map.suspend_bitset(uaddr, process->tgid(), tc, val3);
375  }
376 
377  return 0;
378  } else if (OS::TGT_FUTEX_WAKE == op) {
379  return futex_map.wakeup(uaddr, process->tgid(), val);
380  } else if (OS::TGT_FUTEX_WAKE_BITSET == op) {
381  return futex_map.wakeup_bitset(uaddr, process->tgid(), val3);
382  } else if (OS::TGT_FUTEX_REQUEUE == op ||
383  OS::TGT_FUTEX_CMP_REQUEUE == op) {
384 
385  // Ensure futex system call accessed atomically.
386  BufferArg buf(uaddr, sizeof(int));
387  buf.copyIn(tc->getVirtProxy());
388  int mem_val = *(int*)buf.bufferPtr();
389  /*
390  * For CMP_REQUEUE, the whole operation is only started only if
391  * val3 is still the value of the futex pointed to by uaddr.
392  */
393  if (OS::TGT_FUTEX_CMP_REQUEUE && val3 != mem_val)
394  return -OS::TGT_EWOULDBLOCK;
395  return futex_map.requeue(uaddr, process->tgid(), val, timeout, uaddr2);
396  } else if (OS::TGT_FUTEX_WAKE_OP == op) {
397  /*
398  * The FUTEX_WAKE_OP operation is equivalent to executing the
399  * following code atomically and totally ordered with respect to
400  * other futex operations on any of the two supplied futex words:
401  *
402  * int oldval = *(int *) addr2;
403  * *(int *) addr2 = oldval op oparg;
404  * futex(addr1, FUTEX_WAKE, val, 0, 0, 0);
405  * if (oldval cmp cmparg)
406  * futex(addr2, FUTEX_WAKE, val2, 0, 0, 0);
407  *
408  * (op, oparg, cmp, cmparg are encoded in val3)
409  *
410  * +---+---+-----------+-----------+
411  * |op |cmp| oparg | cmparg |
412  * +---+---+-----------+-----------+
413  * 4 4 12 12 <== # of bits
414  *
415  * reference: http://man7.org/linux/man-pages/man2/futex.2.html
416  *
417  */
418  // get value from simulated-space
419  BufferArg buf(uaddr2, sizeof(int));
420  buf.copyIn(tc->getVirtProxy());
421  int oldval = *(int*)buf.bufferPtr();
422  int newval = oldval;
423  // extract op, oparg, cmp, cmparg from val3
424  int wake_cmparg = val3 & 0xfff;
425  int wake_oparg = (val3 & 0xfff000) >> 12;
426  int wake_cmp = (val3 & 0xf000000) >> 24;
427  int wake_op = (val3 & 0xf0000000) >> 28;
428  if ((wake_op & OS::TGT_FUTEX_OP_ARG_SHIFT) >> 3 == 1)
429  wake_oparg = (1 << wake_oparg);
430  wake_op &= ~OS::TGT_FUTEX_OP_ARG_SHIFT;
431  // perform operation on the value of the second futex
432  if (wake_op == OS::TGT_FUTEX_OP_SET)
433  newval = wake_oparg;
434  else if (wake_op == OS::TGT_FUTEX_OP_ADD)
435  newval += wake_oparg;
436  else if (wake_op == OS::TGT_FUTEX_OP_OR)
437  newval |= wake_oparg;
438  else if (wake_op == OS::TGT_FUTEX_OP_ANDN)
439  newval &= ~wake_oparg;
440  else if (wake_op == OS::TGT_FUTEX_OP_XOR)
441  newval ^= wake_oparg;
442  // copy updated value back to simulated-space
443  *(int*)buf.bufferPtr() = newval;
444  buf.copyOut(tc->getVirtProxy());
445  // perform the first wake-up
446  int woken1 = futex_map.wakeup(uaddr, process->tgid(), val);
447  int woken2 = 0;
448  // calculate the condition of the second wake-up
449  bool is_wake2 = false;
450  if (wake_cmp == OS::TGT_FUTEX_OP_CMP_EQ)
451  is_wake2 = oldval == wake_cmparg;
452  else if (wake_cmp == OS::TGT_FUTEX_OP_CMP_NE)
453  is_wake2 = oldval != wake_cmparg;
454  else if (wake_cmp == OS::TGT_FUTEX_OP_CMP_LT)
455  is_wake2 = oldval < wake_cmparg;
456  else if (wake_cmp == OS::TGT_FUTEX_OP_CMP_LE)
457  is_wake2 = oldval <= wake_cmparg;
458  else if (wake_cmp == OS::TGT_FUTEX_OP_CMP_GT)
459  is_wake2 = oldval > wake_cmparg;
460  else if (wake_cmp == OS::TGT_FUTEX_OP_CMP_GE)
461  is_wake2 = oldval >= wake_cmparg;
462  // perform the second wake-up
463  if (is_wake2)
464  woken2 = futex_map.wakeup(uaddr2, process->tgid(), timeout);
465 
466  return woken1 + woken2;
467  }
468  warn("futex: op %d not implemented; ignoring.", op);
469  return -ENOSYS;
470 }

1158 template <class OS>
1159 SyscallReturn
1160 mremapFunc(SyscallDesc *desc, int callnum, ThreadContext *tc)
1161 {
1162  int index = 0;
1163  auto process = tc->getProcessPtr();
1164  Addr start = process->getSyscallArg(tc, index);
1165  uint64_t old_length = process->getSyscallArg(tc, index);
1166  uint64_t new_length = process->getSyscallArg(tc, index);
1167  uint64_t flags = process->getSyscallArg(tc, index);
1168  uint64_t provided_address = 0;
1169  bool use_provided_address = flags & OS::TGT_MREMAP_FIXED;
1170 
1171  if (use_provided_address)
1172  provided_address = process->getSyscallArg(tc, index);
1173 
1174  if ((start % TheISA::PageBytes != 0) ||
1175  (provided_address % TheISA::PageBytes != 0)) {
1176  warn("mremap failing: arguments not page aligned");
1177  return -EINVAL;
1178  }
1179 
1180  new_length = roundUp(new_length, TheISA::PageBytes);
1181 
1182  if (new_length > old_length) {
1183  std::shared_ptr<MemState> mem_state = process->memState;
1184  Addr mmap_end = mem_state->getMmapEnd();
1185 
1186  if ((start + old_length) == mmap_end &&
1187  (!use_provided_address || provided_address == start)) {
1188  // This case cannot occur when growing downward, as
1189  // start is greater than or equal to mmap_end.
1190  uint64_t diff = new_length - old_length;
1191  process->allocateMem(mmap_end, diff);
1192  mem_state->setMmapEnd(mmap_end + diff);
1193  return start;
1194  } else {
1195  if (!use_provided_address && !(flags & OS::TGT_MREMAP_MAYMOVE)) {
1196  warn("can't remap here and MREMAP_MAYMOVE flag not set\n");
1197  return -ENOMEM;
1198  } else {
1199  uint64_t new_start = provided_address;
1200  if (!use_provided_address) {
1201  new_start = process->mmapGrowsDown() ?
1202  mmap_end - new_length : mmap_end;
1203  mmap_end = process->mmapGrowsDown() ?
1204  new_start : mmap_end + new_length;
1205  mem_state->setMmapEnd(mmap_end);
1206  }
1207 
1208  process->pTable->remap(start, old_length, new_start);
1209  warn("mremapping to new vaddr %08p-%08p, adding %d\n",
1210  new_start, new_start + new_length,
1211  new_length - old_length);
1212  // add on the remaining unallocated pages
1213  process->allocateMem(new_start + old_length,
1214  new_length - old_length,
1215  use_provided_address /* clobber */);
1216  if (use_provided_address &&
1217  ((new_start + new_length > mem_state->getMmapEnd() &&
1218  !process->mmapGrowsDown()) ||
1219  (new_start < mem_state->getMmapEnd() &&
1220  process->mmapGrowsDown()))) {
1221  // something fishy going on here, at least notify the user
1222  // @todo: increase mmap_end?
1223  warn("mmap region limit exceeded with MREMAP_FIXED\n");
1224  }
1225  warn("returning %08p as start\n", new_start);
1226  return new_start;
1227  }
1228  }
1229  } else {
1230  if (use_provided_address && provided_address != start)
1231  process->pTable->remap(start, new_length, provided_address);
1232  process->pTable->unmap(start + new_length, old_length - new_length);
1233  return use_provided_address ? provided_address : start;
1234  }

1739 template <class OS>
1740 SyscallReturn
1741 mmapImpl(SyscallDesc *desc, int num, ThreadContext *tc, bool is_mmap2)
1742 {
1743  int index = 0;
1744  auto p = tc->getProcessPtr();
1745  Addr start = p->getSyscallArg(tc, index);
1746  uint64_t length = p->getSyscallArg(tc, index);
1747  int prot = p->getSyscallArg(tc, index);
1748  int tgt_flags = p->getSyscallArg(tc, index);
1749  int tgt_fd = p->getSyscallArg(tc, index);
1750  int offset = p->getSyscallArg(tc, index);
1751 
1752  if (is_mmap2)
1753  offset *= TheISA::PageBytes;
1754 
1755  if (start & (TheISA::PageBytes - 1) ||
1756  offset & (TheISA::PageBytes - 1) ||
1757  (tgt_flags & OS::TGT_MAP_PRIVATE &&
1758  tgt_flags & OS::TGT_MAP_SHARED) ||
1759  (!(tgt_flags & OS::TGT_MAP_PRIVATE) &&
1760  !(tgt_flags & OS::TGT_MAP_SHARED)) ||
1761  !length) {
1762  return -EINVAL;
1763  }
1764 
1765  if ((prot & PROT_WRITE) && (tgt_flags & OS::TGT_MAP_SHARED)) {
1766  // With shared mmaps, there are two cases to consider:
1767  // 1) anonymous: writes should modify the mapping and this should be
1768  // visible to observers who share the mapping. Currently, it's
1769  // difficult to update the shared mapping because there's no
1770  // structure which maintains information about the which virtual
1771  // memory areas are shared. If that structure existed, it would be
1772  // possible to make the translations point to the same frames.
1773  // 2) file-backed: writes should modify the mapping and the file
1774  // which is backed by the mapping. The shared mapping problem is the
1775  // same as what was mentioned about the anonymous mappings. For
1776  // file-backed mappings, the writes to the file are difficult
1777  // because it requires syncing what the mapping holds with the file
1778  // that resides on the host system. So, any write on a real system
1779  // would cause the change to be propagated to the file mapping at
1780  // some point in the future (the inode is tracked along with the
1781  // mapping). This isn't guaranteed to always happen, but it usually
1782  // works well enough. The guarantee is provided by the msync system
1783  // call. We could force the change through with shared mappings with
1784  // a call to msync, but that again would require more information
1785  // than we currently maintain.
1786  warn("mmap: writing to shared mmap region is currently "
1787  "unsupported. The write succeeds on the target, but it "
1788  "will not be propagated to the host or shared mappings");
1789  }
1790 
1791  length = roundUp(length, TheISA::PageBytes);
1792 
1793  int sim_fd = -1;
1794  uint8_t *pmap = nullptr;
1795  if (!(tgt_flags & OS::TGT_MAP_ANONYMOUS)) {
1796  std::shared_ptr<FDEntry> fdep = (*p->fds)[tgt_fd];
1797 
1798  auto dfdp = std::dynamic_pointer_cast<DeviceFDEntry>(fdep);
1799  if (dfdp) {
1800  EmulatedDriver *emul_driver = dfdp->getDriver();
1801  return emul_driver->mmap(tc, start, length, prot, tgt_flags,
1802  tgt_fd, offset);
1803  }
1804 
1805  auto ffdp = std::dynamic_pointer_cast<FileFDEntry>(fdep);
1806  if (!ffdp)
1807  return -EBADF;
1808  sim_fd = ffdp->getSimFD();
1809 
1810  pmap = (decltype(pmap))mmap(nullptr, length, PROT_READ, MAP_PRIVATE,
1811  sim_fd, offset);
1812 
1813  if (pmap == (decltype(pmap))-1) {
1814  warn("mmap: failed to map file into host address space");
1815  return -errno;
1816  }
1817  }
1818 
1819  // Extend global mmap region if necessary. Note that we ignore the
1820  // start address unless MAP_FIXED is specified.
1821  if (!(tgt_flags & OS::TGT_MAP_FIXED)) {
1822  std::shared_ptr<MemState> mem_state = p->memState;
1823  Addr mmap_end = mem_state->getMmapEnd();
1824 
1825  start = p->mmapGrowsDown() ? mmap_end - length : mmap_end;
1826  mmap_end = p->mmapGrowsDown() ? start : mmap_end + length;
1827 
1828  mem_state->setMmapEnd(mmap_end);
1829  }
1830 
1831  DPRINTF_SYSCALL(Verbose, " mmap range is 0x%x - 0x%x\n",
1832  start, start + length - 1);
1833 
1834  // We only allow mappings to overwrite existing mappings if
1835  // TGT_MAP_FIXED is set. Otherwise it shouldn't be a problem
1836  // because we ignore the start hint if TGT_MAP_FIXED is not set.
1837  int clobber = tgt_flags & OS::TGT_MAP_FIXED;
1838  if (clobber) {
1839  for (auto tc : p->system->threadContexts) {
1840  // If we might be overwriting old mappings, we need to
1841  // invalidate potentially stale mappings out of the TLBs.
1842  tc->getDTBPtr()->flushAll();
1843  tc->getITBPtr()->flushAll();
1844  }
1845  }
1846 
1847  // Allocate physical memory and map it in. If the page table is already
1848  // mapped and clobber is not set, the simulator will issue throw a
1849  // fatal and bail out of the simulation.
1850  p->allocateMem(start, length, clobber);
1851 
1852  // Transfer content into target address space.
1853  PortProxy &tp = tc->getVirtProxy();
1854  if (tgt_flags & OS::TGT_MAP_ANONYMOUS) {
1855  // In general, we should zero the mapped area for anonymous mappings,
1856  // with something like:
1857  // tp.memsetBlob(start, 0, length);
1858  // However, given that we don't support sparse mappings, and
1859  // some applications can map a couple of gigabytes of space
1860  // (intending sparse usage), that can get painfully expensive.
1861  // Fortunately, since we don't properly implement munmap either,
1862  // there's no danger of remapping used memory, so for now all
1863  // newly mapped memory should already be zeroed so we can skip it.
1864  } else {
1865  // It is possible to mmap an area larger than a file, however
1866  // accessing unmapped portions the system triggers a "Bus error"
1867  // on the host. We must know when to stop copying the file from
1868  // the host into the target address space.
1869  struct stat file_stat;
1870  if (fstat(sim_fd, &file_stat) > 0)
1871  fatal("mmap: cannot stat file");
1872 
1873  // Copy the portion of the file that is resident. This requires
1874  // checking both the mmap size and the filesize that we are
1875  // trying to mmap into this space; the mmap size also depends
1876  // on the specified offset into the file.
1877  uint64_t size = std::min((uint64_t)file_stat.st_size - offset,
1878  length);
1879  tp.writeBlob(start, pmap, size);
1880 
1881  // Cleanup the mmap region before exiting this function.
1882  munmap(pmap, length);
1883 
1884  // Maintain the symbol table for dynamic executables.
1885  // The loader will call mmap to map the images into its address
1886  // space and we intercept that here. We can verify that we are
1887  // executing inside the loader by checking the program counter value.
1888  // XXX: with multiprogrammed workloads or multi-node configurations,
1889  // this will not work since there is a single global symbol table.
1890  if (p->interpImage.contains(tc->pcState().instAddr())) {
1891  std::shared_ptr<FDEntry> fdep = (*p->fds)[tgt_fd];
1892  auto ffdp = std::dynamic_pointer_cast<FileFDEntry>(fdep);
1893  ObjectFile *lib = createObjectFile(ffdp->getFileName());
1894 
1895  if (lib) {
1896  lib->loadAllSymbols(debugSymbolTable,
1897  lib->buildImage().minAddr(), start);
1898  }
1899  }
1900 
1901  // Note that we do not zero out the remainder of the mapping. This
1902  // is done by a real system, but it probably will not affect
1903  // execution (hopefully).
1904  }
1905 
1906  return start;
1907 }
