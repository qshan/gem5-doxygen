96 ElfObject::ElfObject(ImageFileDataPtr ifd) : ObjectFile(ifd)
97 {
98  // get a pointer to elf structure
99  elf = elf_memory((char *)const_cast<uint8_t *>(imageData->data()),
100  imageData->len());
101  assert(elf);
102  gelf_getehdr(elf, &ehdr);
103 
104  determineArch();
105  determineOpSys();
106 
107  entry = ehdr.e_entry;
108  _programHeaderCount = ehdr.e_phnum;
109  _programHeaderSize = ehdr.e_phentsize;
110 
111  // Go through all the segments in the program and record them.
112  for (int i = 0; i < ehdr.e_phnum; ++i) {
113  GElf_Phdr phdr;
114  if (gelf_getphdr(elf, i, &phdr) == 0) {
115  panic("gelf_getphdr failed for segment %d.", i);
116  }
117 
118  if (phdr.p_type == PT_LOAD)
119  handleLoadableSegment(phdr, i);
120  if (phdr.p_type == PT_INTERP) {
121  // Make sure the interpreter is an valid ELF file.
122  char *interp_path = (char *)imageData->data() + phdr.p_offset;
123  ObjectFile *obj = createObjectFile(interp_path);
124  interpreter = dynamic_cast<ElfObject *>(obj);
125  assert(interpreter != nullptr);
126  }
127  }
128 
129  // should have found at least one loadable segment
130  warn_if(image.segments().empty(),
131  "No loadable segments in '%s'. ELF file corrupted?\n",
132  imageData->filename());
133 
134  for (auto M5_VAR_USED &seg: image.segments())
135  DPRINTFR(Loader, "%s\n", seg);
136 
137  // We will actually read the sections when we need to load them
138 }

140 void
141 ElfObject::determineArch()
142 {
143  auto &emach = ehdr.e_machine;
144  auto &eclass = ehdr.e_ident[EI_CLASS];
145  auto &edata = ehdr.e_ident[EI_DATA];
146 
147  // Detect the architecture
148  if (emach == EM_SPARC64 || (emach == EM_SPARC && eclass == ELFCLASS64) ||
149  emach == EM_SPARCV9) {
150  arch = SPARC64;
151  } else if (emach == EM_SPARC32PLUS ||
152  (emach == EM_SPARC && eclass == ELFCLASS32)) {
153  arch = SPARC32;
154  } else if (emach == EM_MIPS && eclass == ELFCLASS32) {
155  arch = Mips;
156  if (edata != ELFDATA2LSB) {
157  fatal("The binary you're trying to load is compiled for big "
158  "endian MIPS. gem5\nonly supports little endian MIPS. "
159  "Please recompile your binary.\n");
160  }
161  } else if (emach == EM_X86_64 && eclass == ELFCLASS64) {
162  arch = X86_64;
163  } else if (emach == EM_386 && eclass == ELFCLASS32) {
164  arch = I386;
165  } else if (emach == EM_ARM && eclass == ELFCLASS32) {
166  arch = bits(ehdr.e_entry, 0) ? Thumb : Arm;
167  } else if (emach == EM_AARCH64 && eclass == ELFCLASS64) {
168  arch = Arm64;
169  } else if (emach == EM_RISCV) {
170  arch = (eclass == ELFCLASS64) ? Riscv64 : Riscv32;
171  } else if (emach == EM_PPC && eclass == ELFCLASS32) {
172  arch = Power;
173  if (edata != ELFDATA2MSB) {
174  fatal("The binary you're trying to load is compiled for "
175  "little endian Power.\ngem5 only supports big "
176  "endian Power. Please recompile your binary.\n");
177  }
178  } else if (emach == EM_PPC64) {
179  fatal("The binary you're trying to load is compiled for 64-bit "
180  "Power. M5\n only supports 32-bit Power. Please "
181  "recompile your binary.\n");
182  } else if (eclass == ELFCLASS64) {
183  // Since we don't know how to check for alpha right now, we'll
184  // just assume if it wasn't something else and it's 64 bit, that's
185  // what it must be.
186  arch = Alpha;
187  } else {
188  warn("Unknown architecture: %d\n", emach);
189  }
190 }

192 void
193 ElfObject::determineOpSys()
194 {
195  // Detect the operating system
196  switch (ehdr.e_ident[EI_OSABI]) {
197  case ELFOSABI_LINUX:
198  opSys = Linux;
199  return;
200  case ELFOSABI_SOLARIS:
201  opSys = Solaris;
202  return;
203  case ELFOSABI_TRU64:
204  opSys = Tru64;
205  return;
206  case ELFOSABI_ARM:
207  opSys = LinuxArmOABI;
208  return;
209  case ELFOSABI_FREEBSD:
210  opSys = FreeBSD;
211  return;
212  default:
213  opSys = UnknownOpSys;
214  }
215 
216  Elf_Scn *section = elf_getscn(elf, 1);
217  for (int sec_idx = 1; section; section = elf_getscn(elf, ++sec_idx)) {
218  GElf_Shdr shdr;
219  gelf_getshdr(section, &shdr);
220 
221  char *e_str = elf_strptr(elf, ehdr.e_shstrndx, shdr.sh_name);
222  if (shdr.sh_type == SHT_NOTE && !strcmp(".note.ABI-tag", e_str)) {
223  // we have found a ABI note section
224  // Check the 5th 32bit word for OS 0 == linux, 1 == hurd,
225  // 2 == solaris, 3 == freebsd
226  Elf_Data *raw_data = elf_rawdata(section, nullptr);
227  assert(raw_data && raw_data->d_buf);
228 
229  uint32_t raw_abi = ((uint32_t *)raw_data->d_buf)[4];
230  bool is_le = ehdr.e_ident[EI_DATA] == ELFDATA2LSB;
231  uint32_t os_abi = is_le ? htole(raw_abi) : htobe(raw_abi);
232 
233  switch (os_abi) {
234  case 0:
235  opSys = Linux;
236  return;
237  case 1:
238  fatal("gem5 does not support the HURD ABI.\n");
239  case 2:
240  opSys = Solaris;
241  return;
242  case 3:
243  opSys = FreeBSD;
244  return;
245  }
246  }
247 
248  if (!strcmp(".SUNW_version", e_str) || !strcmp(".stab.index", e_str)) {
249  opSys = Solaris;
250  return;
251  }
252  }
253 }

255 void
256 ElfObject::handleLoadableSegment(GElf_Phdr phdr, int seg_num)
257 {
258  auto name = std::to_string(seg_num);
259 
260  image.addSegment({ name, phdr.p_paddr, imageData,
261  phdr.p_offset, phdr.p_filesz });
262  Addr uninitialized = phdr.p_memsz - phdr.p_filesz;
263  if (uninitialized) {
264  // There may be parts of a segment which aren't included in the
265  // file. In those cases, we need to create a new segment with no
266  // data to take up the extra space. This should be zeroed when
267  // loaded into memory.
268  image.addSegment({ name + "(uninitialized)",
269  phdr.p_paddr + phdr.p_filesz, uninitialized });
270  }
271 
272  const Addr file_start = phdr.p_offset;
273  const Addr file_end = file_start + phdr.p_filesz;
274 
275  // If there is a program header table, figure out the virtual
276  // address of the header table in the final memory image. We use
277  // the program headers themselves to translate from a file offset
278  // to the address in the image.
279  if (file_start <= ehdr.e_phoff && file_end > ehdr.e_phoff)
280  _programHeaderTable = phdr.p_vaddr + (ehdr.e_phoff - file_start);
281 }

288 bool
289 ElfObject::loadSomeSymbols(SymbolTable *symtab, int binding, Addr mask,
290  Addr base, Addr offset)
291 {
292  if (!symtab)
293  return false;
294 
295  // check that header matches library version
296  if (elf_version(EV_CURRENT) == EV_NONE)
297  panic("wrong elf version number!");
298 
299  // get a pointer to elf structure
300  Elf *elf = elf_memory((char *)const_cast<uint8_t *>(
301  imageData->data()), imageData->len());
302  assert(elf != NULL);
303 
304  // Get the first section
305  int sec_idx = 1; // there is a 0 but it is nothing, go figure
306  Elf_Scn *section = elf_getscn(elf, sec_idx);
307 
308  // While there are no more sections
309  bool found = false;
310  while (section != NULL) {
311  GElf_Shdr shdr;
312  gelf_getshdr(section, &shdr);
313 
314  if (shdr.sh_type == SHT_SYMTAB) {
315  found = true;
316  Elf_Data *data = elf_getdata(section, NULL);
317  int count = shdr.sh_size / shdr.sh_entsize;
318  DPRINTF(Loader, "Found Symbol Table, %d symbols present\n", count);
319 
320  // loop through all the symbols, only loading global ones
321  for (int i = 0; i < count; ++i) {
322  GElf_Sym sym;
323  gelf_getsym(data, i, &sym);
324  if (GELF_ST_BIND(sym.st_info) == binding) {
325  char *sym_name =
326  elf_strptr(elf, shdr.sh_link, sym.st_name);
327  if (sym_name && sym_name[0] != '$') {
328  Addr value = sym.st_value - base + offset;
329  if (symtab->insert(value & mask, sym_name)) {
330  DPRINTF(Loader, "Symbol: %-40s value %#x\n",
331  sym_name, value);
332  }
333  }
334  }
335  }
336  }
337  ++sec_idx;
338  section = elf_getscn(elf, sec_idx);
339  }
340 
341  elf_end(elf);
342 
343  return found;
344 }
