52 // -----------------------------------------------------------------------------
53 // CLASS sc_barrier
54 //
55 // This class provides a way of synchronising a set of processes. Each process
56 // calls the wait() method and one all processes have called that method they
57 // will all be released for execution.
58 // -----------------------------------------------------------------------------
59 
60 class sc_barrier {
61  public:
62  void initialize( int thread_n )
63  {
64  m_thread_n = thread_n;
65  }
66 
67  void wait()
68  {
69  m_thread_n--;
70  if ( m_thread_n )
71  {
72  ::sc_core::wait(m_barrier_event);
73  }
74  else
75  {
76  m_barrier_event.notify(SC_ZERO_TIME);
77  ::sc_core::wait(m_barrier_event);
78  }
79  }
80 
81  protected:
82  sc_event m_barrier_event; // Event to wait on.
83  int m_thread_n; // # of threads left to wait.
84 };
