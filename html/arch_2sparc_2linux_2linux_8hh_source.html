189  static void
190  archClone(uint64_t flags,
191  Process *pp, Process *cp,
192  ThreadContext *ptc, ThreadContext *ctc,
193  uint64_t stack, uint64_t tls)
194  {
195  SparcISA::copyRegs(ptc, ctc);
196  ctc->setIntReg(SparcISA::NumIntArchRegs + 6, 0);
197  ctc->setIntReg(SparcISA::NumIntArchRegs + 4, 0);
198  ctc->setIntReg(SparcISA::NumIntArchRegs + 3, SparcISA::NWindows - 2);
199  ctc->setIntReg(SparcISA::NumIntArchRegs + 5, SparcISA::NWindows);
200  ctc->setMiscReg(SparcISA::MISCREG_CWP, 0);
201  ctc->setIntReg(SparcISA::NumIntArchRegs + 7, 0);
202  ctc->setMiscRegNoEffect(SparcISA::MISCREG_TL, 0);
203  ctc->setMiscReg(SparcISA::MISCREG_ASI, SparcISA::ASI_PRIMARY);
204  for (int y = 8; y < 32; y++)
205  ctc->setIntReg(y, ptc->readIntReg(y));
206 
207  if (stack)
208  ctc->setIntReg(SparcISA::StackPointerReg, stack);
209  }
