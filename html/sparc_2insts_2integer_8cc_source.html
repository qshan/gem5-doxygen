57 bool
58 IntOpImm::printPseudoOps(std::ostream &os, Addr pc,

63  if (imm == 0) {
64  printMnemonic(os, "clr");

72  printMnemonic(os, "mov");
73  printSrcReg(os, 0);

75  printDestReg(os, 0);
76  return true;

83 IntOp::generateDisassembly(Addr pc, const SymbolTable *symtab) const
