121 // This class specializes the one above and sets up ThreadContexts based on
122 // its template parameters. These ThreadContexts provide the standard gem5
123 // interface and translate those accesses to use the Iris API to access that
124 // state in the target context.
125 template <class TC>
126 class CPU : public Iris::BaseCPU
127 {
128  public:
129  CPU(IrisBaseCPUParams *params, iris::IrisConnectionInterface *iris_if) :
130  BaseCPU(params, params->evs)
131  {
132  const std::string parent_path = evs->name();
133  System *sys = params->system;
134 
135  int thread_id = 0;
136  for (const std::string &sub_path: params->thread_paths) {
137  std::string path = parent_path + "." + sub_path;
138  auto *tc = new TC(this, thread_id++, sys,
139  params->dtb, params->itb,iris_if, path);
140  threadContexts.push_back(tc);
141  }
142  }
143 };
