145 
146  public:

150 
151  Addr npc() const { return _npc; }
152  void npc(Addr val) { _npc = val; }
153 
154  void
155  set(Addr val)
156  {
157  pc(val);
158  npc(val + sizeof(MachInst));
159  };
160 
161  void
162  setNPC(Addr val)
163  {
164  npc(val);
165  }
166 
167  SimplePCState() {}
168  SimplePCState(Addr val) { set(val); }
169 
170  bool
171  branching() const
172  {
173  return this->npc() != this->pc() + sizeof(MachInst);
174  }
175 
176  // Advance the PC.
177  void
178  advance()
179  {
180  _pc = _npc;
181  _npc += sizeof(MachInst);
