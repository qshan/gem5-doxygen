64 SimpleDisk::read(Addr addr, baddr_t block, int count) const
65 {
66  uint8_t *data = new uint8_t[SectorSize * count];
67 
68  if (count & (SectorSize - 1))
69  panic("Not reading a multiple of a sector (count = %d)", count);
70 
71  for (int i = 0, j = 0; i < count; i += SectorSize, j++)
72  image->read(data + i, block + j);
73 
74  system->physProxy.writeBlob(addr, data, count);
75 
76  DPRINTF(SimpleDisk, "read block=%#x len=%d\n", (uint64_t)block, count);
77  DDUMP(SimpleDiskData, data, count);
78 
79  delete [] data;
80 }
