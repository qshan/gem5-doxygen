75 PixelConverter::readWord(const uint8_t *p) const
76 {
77  uint32_t word(0);
78 
79  if (byte_order == LittleEndianByteOrder) {
80  for (int i = 0; i < length; ++i)
81  word |= p[i] << (8 * i);
82  } else {
83  for (int i = 0; i < length; ++i)
84  word |= p[i] << (8 * (length - i - 1));
