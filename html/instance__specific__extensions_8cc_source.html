55  unsigned int
56  register_extension(std::type_index type)
57  {
58  type_map::const_iterator it = ids_.find(type);
59 
60  if (it == ids_.end()) {
61  // New extension - generate/store ID.
62  type_map::value_type v(type, static_cast<key_type>(ids_.size()));
63  ids_.insert(v);
64  return v.second;
65  }
66  return it->second;
67  }
