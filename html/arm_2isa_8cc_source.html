217 void
218 ISA::clear32(const ArmISAParams *p, const SCTLR &sctlr_rst)
219 {
220  CPSR cpsr = 0;
221  cpsr.mode = MODE_USER;
222 
223  if (FullSystem) {
224  miscRegs[MISCREG_MVBAR] = system->resetAddr();
225  }
226 
227  miscRegs[MISCREG_CPSR] = cpsr;
228  updateRegMap(cpsr);
229 
230  SCTLR sctlr = 0;
231  sctlr.te = (bool) sctlr_rst.te;
232  sctlr.nmfi = (bool) sctlr_rst.nmfi;
233  sctlr.v = (bool) sctlr_rst.v;
234  sctlr.u = 1;
235  sctlr.xp = 1;
236  sctlr.rao2 = 1;
237  sctlr.rao3 = 1;
238  sctlr.rao4 = 0xf; // SCTLR[6:3]
239  sctlr.uci = 1;
240  sctlr.dze = 1;
241  miscRegs[MISCREG_SCTLR_NS] = sctlr;
242  miscRegs[MISCREG_SCTLR_RST] = sctlr_rst;
243  miscRegs[MISCREG_HCPTR] = 0;
244 
245  miscRegs[MISCREG_CPACR] = 0;
246 
247  miscRegs[MISCREG_FPSID] = p->fpsid;
248 
249  if (haveLPAE) {
250  TTBCR ttbcr = miscRegs[MISCREG_TTBCR_NS];
251  ttbcr.eae = 0;
252  miscRegs[MISCREG_TTBCR_NS] = ttbcr;
253  // Enforce consistency with system-level settings
254  miscRegs[MISCREG_ID_MMFR0] = (miscRegs[MISCREG_ID_MMFR0] & ~0xf) | 0x5;
255  }
256 
257  if (haveSecurity) {
258  miscRegs[MISCREG_SCTLR_S] = sctlr;
259  miscRegs[MISCREG_SCR] = 0;
260  miscRegs[MISCREG_VBAR_S] = 0;
261  } else {
262  // we're always non-secure
263  miscRegs[MISCREG_SCR] = 1;
264  }
265 
266  //XXX We need to initialize the rest of the state.
267 }

269 void
270 ISA::clear64(const ArmISAParams *p)
271 {
272  CPSR cpsr = 0;
273  Addr rvbar = system->resetAddr();
274  switch (system->highestEL()) {
275  // Set initial EL to highest implemented EL using associated stack
276  // pointer (SP_ELx); set RVBAR_ELx to implementation defined reset
277  // value
278  case EL3:
279  cpsr.mode = MODE_EL3H;
280  miscRegs[MISCREG_RVBAR_EL3] = rvbar;
281  break;
282  case EL2:
283  cpsr.mode = MODE_EL2H;
284  miscRegs[MISCREG_RVBAR_EL2] = rvbar;
285  break;
286  case EL1:
287  cpsr.mode = MODE_EL1H;
288  miscRegs[MISCREG_RVBAR_EL1] = rvbar;
289  break;
290  default:
291  panic("Invalid highest implemented exception level");
292  break;
293  }
294 
295  // Initialize rest of CPSR
296  cpsr.daif = 0xf; // Mask all interrupts
297  cpsr.ss = 0;
298  cpsr.il = 0;
299  miscRegs[MISCREG_CPSR] = cpsr;
300  updateRegMap(cpsr);
301 
302  // Initialize other control registers
303  miscRegs[MISCREG_MPIDR_EL1] = 0x80000000;
304  if (haveSecurity) {
305  miscRegs[MISCREG_SCTLR_EL3] = 0x30c50830;
306  miscRegs[MISCREG_SCR_EL3] = 0x00000030; // RES1 fields
307  } else if (haveVirtualization) {
308  // also MISCREG_SCTLR_EL2 (by mapping)
309  miscRegs[MISCREG_HSCTLR] = 0x30c50830;
310  } else {
311  // also MISCREG_SCTLR_EL1 (by mapping)
312  miscRegs[MISCREG_SCTLR_NS] = 0x30d00800 | 0x00050030; // RES1 | init
313  // Always non-secure
314  miscRegs[MISCREG_SCR_EL3] = 1;
315  }
316 }

430 RegVal
431 ISA::readMiscRegNoEffect(int misc_reg) const
432 {
433  assert(misc_reg < NumMiscRegs);
434 
435  const auto &reg = lookUpMiscReg[misc_reg]; // bit masks
436  const auto &map = getMiscIndices(misc_reg);
437  int lower = map.first, upper = map.second;
438  // NB!: apply architectural masks according to desired register,
439  // despite possibly getting value from different (mapped) register.
440  auto val = !upper ? miscRegs[lower] : ((miscRegs[lower] & mask(32))
441  |(miscRegs[upper] << 32));
442  if (val & reg.res0()) {
443  DPRINTF(MiscRegs, "Reading MiscReg %s with set res0 bits: %#x\n",
444  miscRegName[misc_reg], val & reg.res0());
445  }
446  if ((val & reg.res1()) != reg.res1()) {
447  DPRINTF(MiscRegs, "Reading MiscReg %s with clear res1 bits: %#x\n",
448  miscRegName[misc_reg], (val & reg.res1()) ^ reg.res1());
449  }
450  return (val & ~reg.raz()) | reg.rao(); // enforce raz/rao
451 }

454 RegVal
455 ISA::readMiscReg(int misc_reg, ThreadContext *tc)
456 {
457  CPSR cpsr = 0;
458  PCState pc = 0;
459  SCR scr = 0;
460 
461  if (misc_reg == MISCREG_CPSR) {
462  cpsr = miscRegs[misc_reg];
463  pc = tc->pcState();
464  cpsr.j = pc.jazelle() ? 1 : 0;
465  cpsr.t = pc.thumb() ? 1 : 0;
466  return cpsr;
467  }
468 
469 #ifndef NDEBUG
470  if (!miscRegInfo[misc_reg][MISCREG_IMPLEMENTED]) {
471  if (miscRegInfo[misc_reg][MISCREG_WARN_NOT_FAIL])
472  warn("Unimplemented system register %s read.\n",
473  miscRegName[misc_reg]);
474  else
475  panic("Unimplemented system register %s read.\n",
476  miscRegName[misc_reg]);
477  }
478 #endif
479 
480  switch (unflattenMiscReg(misc_reg)) {
481  case MISCREG_HCR:
482  case MISCREG_HCR2:
483  if (!haveVirtualization)
484  return 0;
485  break;
486  case MISCREG_CPACR:
487  {
488  const uint32_t ones = (uint32_t)(-1);
489  CPACR cpacrMask = 0;
490  // Only cp10, cp11, and ase are implemented, nothing else should
491  // be readable? (straight copy from the write code)
492  cpacrMask.cp10 = ones;
493  cpacrMask.cp11 = ones;
494  cpacrMask.asedis = ones;
495 
496  // Security Extensions may limit the readability of CPACR
497  if (haveSecurity) {
498  scr = readMiscRegNoEffect(MISCREG_SCR);
499  cpsr = readMiscRegNoEffect(MISCREG_CPSR);
500  if (scr.ns && (cpsr.mode != MODE_MON) && ELIs32(tc, EL3)) {
501  NSACR nsacr = readMiscRegNoEffect(MISCREG_NSACR);
502  // NB: Skipping the full loop, here
503  if (!nsacr.cp10) cpacrMask.cp10 = 0;
504  if (!nsacr.cp11) cpacrMask.cp11 = 0;
505  }
506  }
507  RegVal val = readMiscRegNoEffect(MISCREG_CPACR);
508  val &= cpacrMask;
509  DPRINTF(MiscRegs, "Reading misc reg %s: %#x\n",
510  miscRegName[misc_reg], val);
511  return val;
512  }
513  case MISCREG_MPIDR:
514  case MISCREG_MPIDR_EL1:
515  return readMPIDR(system, tc);
516  case MISCREG_VMPIDR:
517  case MISCREG_VMPIDR_EL2:
518  // top bit defined as RES1
519  return readMiscRegNoEffect(misc_reg) | 0x80000000;
520  case MISCREG_ID_AFR0: // not implemented, so alias MIDR
521  case MISCREG_REVIDR: // not implemented, so alias MIDR
522  case MISCREG_MIDR:
523  cpsr = readMiscRegNoEffect(MISCREG_CPSR);
524  scr = readMiscRegNoEffect(MISCREG_SCR);
525  if ((cpsr.mode == MODE_HYP) || inSecureState(scr, cpsr)) {
526  return readMiscRegNoEffect(misc_reg);
527  } else {
528  return readMiscRegNoEffect(MISCREG_VPIDR);
529  }
530  break;
531  case MISCREG_JOSCR: // Jazelle trivial implementation, RAZ/WI
532  case MISCREG_JMCR: // Jazelle trivial implementation, RAZ/WI
533  case MISCREG_JIDR: // Jazelle trivial implementation, RAZ/WI
534  case MISCREG_AIDR: // AUX ID set to 0
535  case MISCREG_TCMTR: // No TCM's
536  return 0;
537 
538  case MISCREG_CLIDR:
539  warn_once("The clidr register always reports 0 caches.\n");
540  warn_once("clidr LoUIS field of 0b001 to match current "
541  "ARM implementations.\n");
542  return 0x00200000;
543  case MISCREG_CCSIDR:
544  warn_once("The ccsidr register isn't implemented and "
545  "always reads as 0.\n");
546  break;
547  case MISCREG_CTR: // AArch32, ARMv7, top bit set
548  case MISCREG_CTR_EL0: // AArch64
549  {
550  //all caches have the same line size in gem5
551  //4 byte words in ARM
552  unsigned lineSizeWords =
553  tc->getSystemPtr()->cacheLineSize() / 4;
554  unsigned log2LineSizeWords = 0;
555 
556  while (lineSizeWords >>= 1) {
557  ++log2LineSizeWords;
558  }
559 
560  CTR ctr = 0;
561  //log2 of minimun i-cache line size (words)
562  ctr.iCacheLineSize = log2LineSizeWords;
563  //b11 - gem5 uses pipt
564  ctr.l1IndexPolicy = 0x3;
565  //log2 of minimum d-cache line size (words)

761 void
762 ISA::setMiscRegNoEffect(int misc_reg, RegVal val)
763 {
764  assert(misc_reg < NumMiscRegs);
765 
766  const auto &reg = lookUpMiscReg[misc_reg]; // bit masks
767  const auto &map = getMiscIndices(misc_reg);
768  int lower = map.first, upper = map.second;
769 
770  auto v = (val & ~reg.wi()) | reg.rao();
771  if (upper > 0) {
772  miscRegs[lower] = bits(v, 31, 0);
773  miscRegs[upper] = bits(v, 63, 32);
774  DPRINTF(MiscRegs, "Writing to misc reg %d (%d:%d) : %#x\n",
775  misc_reg, lower, upper, v);
776  } else {
777  miscRegs[lower] = v;
778  DPRINTF(MiscRegs, "Writing to misc reg %d (%d) : %#x\n",
779  misc_reg, lower, v);
780  }
