45 std::string
46 MrsOp::generateDisassembly(Addr pc, const SymbolTable *symtab) const
47 {
48  std::stringstream ss;
49  printMnemonic(ss);
50  printIntReg(ss, dest);
51  ss << ", ";
52  bool foundPsr = false;
53  for (unsigned i = 0; i < numSrcRegs(); i++) {
54  const RegId& reg = srcRegIdx(i);
55  if (!reg.isMiscReg()) {
56  continue;
57  }
58  if (reg.index() == MISCREG_CPSR) {
59  ss << "cpsr";
60  foundPsr = true;
61  break;
62  }
63  if (reg.index() == MISCREG_SPSR) {
64  ss << "spsr";
65  foundPsr = true;
66  break;
67  }
68  }
69  if (!foundPsr) {
70  ss << "????";
71  }
72  return ss.str();
73 }

75 void
76 MsrBase::printMsrBase(std::ostream &os) const
77 {
78  printMnemonic(os);
79  bool apsr = false;
80  bool foundPsr = false;
81  for (unsigned i = 0; i < numDestRegs(); i++) {
82  const RegId& reg = destRegIdx(i);
83  if (!reg.isMiscReg()) {
84  continue;
85  }
86  if (reg.index() == MISCREG_CPSR) {
87  os << "cpsr_";
88  foundPsr = true;
89  break;
90  }
91  if (reg.index() == MISCREG_SPSR) {
92  if (bits(byteMask, 1, 0)) {
93  os << "spsr_";
94  } else {
95  os << "apsr_";
96  apsr = true;
97  }
98  foundPsr = true;
99  break;
100  }
101  }
102  if (!foundPsr) {
103  os << "????";
104  return;
105  }
106  if (bits(byteMask, 3)) {
107  if (apsr) {
108  os << "nzcvq";
109  } else {
110  os << "f";
111  }
112  }
113  if (bits(byteMask, 2)) {
114  if (apsr) {
115  os << "g";
116  } else {
117  os << "s";
118  }
119  }
120  if (bits(byteMask, 1)) {
121  os << "x";
122  }
123  if (bits(byteMask, 0)) {
124  os << "c";
125  }
126 }
