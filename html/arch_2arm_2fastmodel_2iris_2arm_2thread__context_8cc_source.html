62 TheISA::PCState
63 ArmThreadContext::pcState() const
64 {
65  ArmISA::CPSR cpsr = readMiscRegNoEffect(ArmISA::MISCREG_CPSR);
66  ArmISA::PCState pc;
67 
68  pc.thumb(cpsr.t);
69  pc.nextThumb(pc.thumb());
70  pc.jazelle(cpsr.j);
71  pc.nextJazelle(cpsr.j);
72  pc.aarch64(!cpsr.width);
73  pc.nextAArch64(!cpsr.width);
74  pc.illegalExec(false);
75 
76  iris::ResourceReadResult result;
77  call().resource_read(_instId, result, pcRscId);
78  Addr addr = result.data.at(0);
79  if (cpsr.width && cpsr.t)
80  addr = addr & ~0x1;
81  pc.set(addr);
82 
83  return pc;
84 }
85 void
86 ArmThreadContext::pcState(const TheISA::PCState &val)
87 {
88  Addr pc = val.pc();
89 
90  ArmISA::CPSR cpsr = readMiscRegNoEffect(ArmISA::MISCREG_CPSR);
91  if (cpsr.width && cpsr.t)
92  pc = pc | 0x1;
93 
94  iris::ResourceWriteResult result;
95  call().resource_write(_instId, result, pcRscId, pc);
96 }

110 uint64_t
111 ArmThreadContext::readIntReg(RegIndex reg_idx) const

135 const ArmISA::VecRegContainer &
136 ArmThreadContext::readVecReg(const RegId &reg_id) const
137 {
138  const RegIndex idx = reg_id.index();
139  ArmISA::VecRegContainer &reg = vecRegs.at(idx);
140 
141  iris::ResourceReadResult result;
142  call().resource_read(_instId, result, vecRegIds.at(idx));
143  size_t data_size = result.data.size() * (sizeof(*result.data.data()));
144  size_t size = std::min(data_size, reg.SIZE);
145  memcpy(reg.raw_ptr<void>(), (void *)result.data.data(), size);
146 
147  return reg;
148 }
