67 MasterPort::bind(Port &peer)
68 {
69  auto *slave_port = dynamic_cast<SlavePort *>(&peer);
70  if (!slave_port) {
71  fatal("Attempt to bind port %s to non-slave port %s.",
72  name(), peer.name());
73  }
74  // master port keeps track of the slave port
75  _slavePort = slave_port;
76  Port::bind(peer);
77  // slave port also keeps track of master port
78  _slavePort->slaveBind(*this);
79 }
