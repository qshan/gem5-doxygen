90 #define ECOFF_ROUND(value, by) \
91  (((value) + (by) - 1) & ~((by) - 1))

93 #define ECOFF_BLOCK_ALIGN(ep, value) \
94  ((ep)->a.magic == ECOFF_ZMAGIC ? ECOFF_ROUND((value), ECOFF_LDPGSZ) : \
95  (value))

97 #define ECOFF_TXTOFF(ep) \
98  ((ep)->a.magic == ECOFF_ZMAGIC ? 0 : \
99  ECOFF_ROUND(ECOFF_HDR_SIZE + (ep)->f.f_nscns * \
100  sizeof(struct ecoff_scnhdr), ECOFF_SEGMENT_ALIGNMENT(ep)))
101 
102 #define ECOFF_DATOFF(ep) \

105 #define ECOFF_SEGMENT_ALIGN(ep, value) \
