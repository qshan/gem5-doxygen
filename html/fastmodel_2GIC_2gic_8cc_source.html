49 SCGIC::Terminator::Terminator(
50  sc_core::sc_module_name _name, Initiators &inits) :
51  sc_core::sc_module(_name),
52  targets("targets", countUnbound(inits))
53 {
54  // For every unbound initiator socket, connected it to one
55  // terminator target socket.
56  int i = 0;
57  for (auto &init: inits) {
58  if (!init.get_port_base().size()) {
59  auto &term = targets.at(i++);
60  term.bind(*this);
61  term.bind(init);
62  }
63  }
64 }
