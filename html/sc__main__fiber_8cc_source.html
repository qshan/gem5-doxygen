49 ScMainFiber::main()
50 {
51  _called = true;
52 
53  if (::sc_main) {
54  try {
55  _resultInt = ::sc_main(_argc, _argv);
56  if (_resultInt)
57  _resultStr = "sc_main returned non-zero";
58  else
59  _resultStr = "sc_main finished";
60  // Make sure no systemc events/notifications are scheduled
61  // after sc_main returns.
62  } catch (const ::sc_core::sc_report &r) {
63  // There was an exception nobody caught.
64  _resultStr = "uncaught sc_report";
65  reportHandlerProc(
66  r, ::sc_core::sc_report_handler::get_catch_actions());
67  } catch (...) {
68  // There was some other type of exception we need to wrap.
69  _resultStr = "uncaught exception";
70  reportHandlerProc(reportifyException(),
71  ::sc_core::sc_report_handler::get_catch_actions());
72  }
73  scheduler.clear();
74  } else {
75  // If python tries to call sc_main but no sc_main was defined...
76  fatal("sc_main called but not defined.\n");
77  }
78 }
