53  static Wavefront*
54  __chooseWave(std::vector<Wavefront*> *sched_list)
55  {
56  panic_if(!sched_list->size(), "OF scheduling policy sched list is "
57  "empty.\n");
58  // set when policy choose a wave to schedule
59  Wavefront *selected_wave(nullptr);
60  int selected_wave_id = -1;
61  int selected_position = 0;
62 
63  for (int position = 0; position < sched_list->size(); ++position) {
64  Wavefront *cur_wave = sched_list->at(position);
65  int cur_wave_id = cur_wave->wfDynId;
66 
67  // Choosed wave with the lowest wave ID
68  if (selected_wave_id == -1 || cur_wave_id < selected_wave_id) {
69  selected_wave_id = cur_wave_id;
70  selected_wave = cur_wave;
71  selected_position = position;
72  }
73  }
74 
75  // Check to make sure ready list had at least one schedulable wave
76  panic_if(!selected_wave, "No wave found by OF scheduling policy.\n");
77  sched_list->erase(sched_list->begin() + selected_position);
78 
79  return selected_wave;
80  }
81 };
