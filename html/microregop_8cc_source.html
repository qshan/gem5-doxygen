50  uint64_t RegOpBase::genFlags(uint64_t oldFlags, uint64_t flagMask,
51  uint64_t _dest, uint64_t _src1, uint64_t _src2,
52  bool subtract) const
53  {
54  DPRINTF(X86, "flagMask = %#x\n", flagMask);
55  uint64_t flags = oldFlags & ~flagMask;
56  if (flagMask & (ECFBit | CFBit))
57  {
58  if (findCarry(dataSize*8, _dest, _src1, _src2))
59  flags |= (flagMask & (ECFBit | CFBit));
60  if (subtract)
61  flags ^= (flagMask & (ECFBit | CFBit));
62  }
63  if (flagMask & PFBit && !findParity(8, _dest))
64  flags |= PFBit;
65  if (flagMask & AFBit)
66  {
67  if (findCarry(4, _dest, _src1, _src2))
68  flags |= AFBit;
69  if (subtract)
70  flags ^= AFBit;
71  }
72  if (flagMask & (EZFBit | ZFBit) && findZero(dataSize*8, _dest))
73  flags |= (flagMask & (EZFBit | ZFBit));
74  if (flagMask & SFBit && findNegative(dataSize*8, _dest))
75  flags |= SFBit;
76  if (flagMask & OFBit && findOverflow(dataSize*8, _dest, _src1, _src2))
77  flags |= OFBit;
78  return flags;
79  }
