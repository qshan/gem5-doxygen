71 void
72 ThreadContext::maintainStepping()
73 {
74  Tick now = 0;
75 
76  while (true) {
77  if (comInstEventQueue.empty()) {
78  // Set to 0 to deactivate stepping.
79  call().step_setup(_instId, 0, "instruction");
80  break;
81  }
82 
83  Tick next = comInstEventQueue.nextTick();
84  if (!now)
85  now = getCurrentInstCount();
86 
87  if (next <= now) {
88  comInstEventQueue.serviceEvents(now);
89  // Start over now that comInstEventQueue has likely changed.
90  continue;
91  }
92 
93  // Set to the number of instructions still to step through.
94  Tick remaining = next - now;
95  call().step_setup(_instId, remaining, "instruction");
96  break;
97  }
98 }

100 iris::IrisErrorCode
101 ThreadContext::instanceRegistryChanged(
102  uint64_t esId, const iris::IrisValueMap &fields, uint64_t time,
103  uint64_t sInstId, bool syncEc, std::string &error_message_out)
104 {
105  const std::string &event = fields.at("EVENT").getString();
106  const iris::InstanceId id = fields.at("INST_ID").getU64();
107  const std::string &name = fields.at("INST_NAME").getString();
108 
109  if (name != "component." + _irisPath)
110  return iris::E_ok;
111 
112  if (event == "added")
113  _instId = id;
114  else if (event == "removed")
115  _instId = iris::IRIS_UINT64_MAX;
116  else
117  panic("Unrecognized event type %s", event);
118 
119  return iris::E_ok;
120 }

139 iris::IrisErrorCode
140 ThreadContext::simulationTimeEvent(
141  uint64_t esId, const iris::IrisValueMap &fields, uint64_t time,
142  uint64_t sInstId, bool syncEc, std::string &error_message_out)
143 {
144  if (fields.at("RUNNING").getAsBool()) {
145  // If this is just simulation time starting up, don't do anything.
146  return iris::E_ok;
147  }
148 
149  // If simulation time has stopped for any reason, IRIS helpfully clears
150  // all stepping counters and we need to set them back. We might also need
151  // to service events based on the current number of executed instructions.
152  maintainStepping();
153 
154  // Restart simulation time to make sure things progress once we give
155  // control back.
156  call().simulationTime_run(iris::IrisInstIdSimulationEngine);
157 
158  return iris::E_ok;
159 }

161 ThreadContext::ThreadContext(
162  BaseCPU *cpu, int id, System *system, ::BaseTLB *dtb, ::BaseTLB *itb,
163  iris::IrisConnectionInterface *iris_if, const std::string &iris_path) :
164  _cpu(cpu), _threadId(id), _system(system), _dtb(dtb), _itb(itb),
165  _irisPath(iris_path), _instId(iris::IRIS_UINT64_MAX), _status(Active),
166  comInstEventQueue("instruction-based event queue"),
167  client(iris_if, "client." + iris_path)
168 {
169  iris::InstanceInfo info;
170  auto ret_code = noThrow().instanceRegistry_getInstanceInfoByName(
171  info, "component." + iris_path);
172  if (ret_code == iris::E_ok) {
173  // The iris instance registry already new about this path.
174  _instId = info.instId;
175  } else {
176  // This path doesn't (yet) exist. Set the ID to something invalid.
177  _instId = iris::IRIS_UINT64_MAX;
178  }

194  client.registerEventCallback<Self, &Self::phaseInitLeave>(

200  call().eventStream_create(
201  iris::IrisInstIdSimulationEngine, initEventStreamId,
202  evSrcInfo.evSrcId, client.getInstId());
203 
204  client.registerEventCallback<Self, &Self::simulationTimeEvent>(
205  this, "ec_IRIS_SIMULATION_TIME_EVENT",

234 void
235 ThreadContext::scheduleInstCountEvent(Event *event, Tick count)
236 {
237  Tick now = getCurrentInstCount();
238  comInstEventQueue.schedule(event, count);
239  if (count <= now)
240  call().simulationTime_stop(iris::IrisInstIdSimulationEngine);
241  else
242  maintainStepping();
