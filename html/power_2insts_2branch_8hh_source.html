131 class BranchCond : public PCDependentDisassembly
132 {
133  protected:
134 
136  uint32_t bo;
137  uint32_t bi;
138 
140  BranchCond(const char *mnem, MachInst _machInst, OpClass __opClass)
141  : PCDependentDisassembly(mnem, _machInst, __opClass),
142  bo(machInst.bo),
143  bi(machInst.bi)
144  {
145  }
146 
147  inline bool
148  ctrOk(uint32_t& ctr) const
149  {
150  bool ctr_ok;
151  if (bo & 4) {
152  ctr_ok = true;
153  } else {
154  ctr--;
155  if (ctr != 0) {
156  ctr_ok = ((bo & 2) == 0);
157  } else {
158  ctr_ok = ((bo & 2) != 0);
159  }
160  }
161  return ctr_ok;
162  }
163 
164  inline bool
165  condOk(uint32_t cr) const
166  {
167  bool cond_ok;
168  if (bo & 16) {
169  cond_ok = true;
170  } else {
171  cond_ok = (((cr >> (31 - bi)) & 1) == ((bo >> 3) & 1));
172  }
173  return cond_ok;
174  }
175 };
