36 // This is designed for situations where we have a pointer to a base
37 // type, but in all cases when we cast it to a derived type, we know
38 // by construction that it should work correctly.

41 
42 // In debug builds, do the dynamic cast and assert the result is good
43 
44 template <class T, class U>
45 inline T
46 safe_cast(U ptr)
47 {
48  T ret = dynamic_cast<T>(ptr);
49  assert(ret);
50  return ret;
51 }

55 // In non debug builds statically cast the result to the pointer we
