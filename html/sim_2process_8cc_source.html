176 void
177 Process::clone(ThreadContext *otc, ThreadContext *ntc,
178  Process *np, RegVal flags)
179 {
180 #ifndef CLONE_VM
181 #define CLONE_VM 0
182 #endif
183 #ifndef CLONE_FILES
184 #define CLONE_FILES 0
185 #endif
186 #ifndef CLONE_THREAD
187 #define CLONE_THREAD 0
188 #endif
189  if (CLONE_VM & flags) {
195  delete np->pTable;
196  np->pTable = pTable;
197  auto &proxy = dynamic_cast<SETranslatingPortProxy &>(
198  ntc->getVirtProxy());
199  proxy.setPageTable(np->pTable);
200 
201  np->memState = memState;
202  } else {
207  typedef std::vector<pair<Addr,Addr>> MapVec;
208  MapVec mappings;
209  pTable->getMappings(&mappings);
210 
211  for (auto map : mappings) {
212  Addr paddr, vaddr = map.first;
213  bool alloc_page = !(np->pTable->translate(vaddr, paddr));
214  np->replicatePage(vaddr, paddr, otc, ntc, alloc_page);
215  }
216 
217  *np->memState = *memState;
218  }
219 
220  if (CLONE_FILES & flags) {
226  np->fds = fds;
227  } else {
235  std::shared_ptr<FDArray> nfds = np->fds;
236  for (int tgt_fd = 0; tgt_fd < fds->getSize(); tgt_fd++) {
237  std::shared_ptr<FDEntry> this_fde = (*fds)[tgt_fd];
238  if (!this_fde) {
239  nfds->setFDEntry(tgt_fd, nullptr);
240  continue;
241  }
242  nfds->setFDEntry(tgt_fd, this_fde->clone());
243 
244  auto this_hbfd = std::dynamic_pointer_cast<HBFDEntry>(this_fde);
245  if (!this_hbfd)
246  continue;
247 
248  int this_sim_fd = this_hbfd->getSimFD();
249  if (this_sim_fd <= 2)
250  continue;
251 
252  int np_sim_fd = dup(this_sim_fd);
253  assert(np_sim_fd != -1);
254 
255  auto nhbfd = std::dynamic_pointer_cast<HBFDEntry>((*nfds)[tgt_fd]);
256  nhbfd->setSimFD(np_sim_fd);
257  }
258  }
259 
260  if (CLONE_THREAD & flags) {
261  np->_tgid = _tgid;
262  delete np->exitGroup;
263  np->exitGroup = exitGroup;
264  }
265 
266  np->argv.insert(np->argv.end(), argv.begin(), argv.end());
267  np->envp.insert(np->envp.end(), envp.begin(), envp.end());
268 }

352 void
353 Process::replicatePage(Addr vaddr, Addr new_paddr, ThreadContext *old_tc,
354  ThreadContext *new_tc, bool allocate_page)
355 {
356  if (allocate_page)
357  new_paddr = system->allocPhysPages(1);
358 
359  // Read from old physical page.
360  uint8_t *buf_p = new uint8_t[PageBytes];
361  old_tc->getVirtProxy().readBlob(vaddr, buf_p, PageBytes);
362 
363  // Create new mapping in process address space by clobbering existing
364  // mapping (if any existed) and then write to the new physical page.
365  bool clobber = true;
366  pTable->map(vaddr, new_paddr, PageBytes, clobber);
367  new_tc->getVirtProxy().writeBlob(vaddr, buf_p, PageBytes);
368  delete[] buf_p;
369 }

371 bool
372 Process::fixupStackFault(Addr vaddr)
373 {
374  Addr stack_min = memState->getStackMin();
375  Addr stack_base = memState->getStackBase();
376  Addr max_stack_size = memState->getMaxStackSize();
377 
378  // Check if this is already on the stack and there's just no page there
379  // yet.
380  if (vaddr >= stack_min && vaddr < stack_base) {
381  allocateMem(roundDown(vaddr, PageBytes), PageBytes);
382  return true;
383  }
384 
385  // We've accessed the next page of the stack, so extend it to include
386  // this address.
387  if (vaddr < stack_min && vaddr >= stack_base - max_stack_size) {
388  while (vaddr < stack_min) {
389  stack_min -= TheISA::PageBytes;
390  if (stack_base - stack_min > max_stack_size)
391  fatal("Maximum stack size exceeded\n");
392  allocateMem(stack_min, TheISA::PageBytes);
393  inform("Increasing stack size by one page.");
394  }
395  memState->setStackMin(stack_min);
396  return true;
397  }
398  return false;
399 }

468 std::string
469 Process::checkPathRedirect(const std::string &filename)
470 {
471  // If the input parameter contains a relative path, convert it.
472  // The target version of the current working directory is fine since
473  // we immediately convert it using redirect paths into a host version.
474  auto abs_path = absolutePath(filename, false);
475 
476  for (auto path : system->redirectPaths) {
477  // Search through the redirect paths to see if a starting substring of
478  // our path falls into any buckets which need to redirected.
479  if (startswith(abs_path, path->appPath())) {
480  std::string tail = abs_path.substr(path->appPath().size());
481 
482  // If this path needs to be redirected, search through a list
483  // of targets to see if we can match a valid file (or directory).
484  for (auto host_path : path->hostPaths()) {
485  if (access((host_path + tail).c_str(), R_OK) == 0) {
486  // Return the valid match.
487  return host_path + tail;
488  }
489  }
490  // The path needs to be redirected, but the file or directory
491  // does not exist on the host filesystem. Return the first
492  // host path as a default.
493  return path->hostPaths()[0] + tail;
494  }
495  }
496 
497  // The path does not need to be redirected.
498  return abs_path;
499 }

501 void
502 Process::updateBias()
503 {
504  ObjectFile *interp = objFile->getInterpreter();
505 
506  if (!interp || !interp->relocatable())
507  return;
508 
509  // Determine how large the interpreters footprint will be in the process
510  // address space.
511  Addr interp_mapsize = roundUp(interp->mapSize(), TheISA::PageBytes);
512 
513  // We are allocating the memory area; set the bias to the lowest address
514  // in the allocated memory region.
515  Addr mmap_end = memState->getMmapEnd();
516  Addr ld_bias = mmapGrowsDown() ? mmap_end - interp_mapsize : mmap_end;
517 
518  // Adjust the process mmap area to give the interpreter room; the real
519  // execve system call would just invoke the kernel's internal mmap
520  // functions to make these adjustments.
521  mmap_end = mmapGrowsDown() ? ld_bias : mmap_end + interp_mapsize;
522  memState->setMmapEnd(mmap_end);
523 
524  interp->updateBias(ld_bias);
525 }

549 std::string
550 Process::absolutePath(const std::string &filename, bool host_filesystem)
551 {
552  if (filename.empty() || startswith(filename, "/"))
553  return filename;
554 
555  // Construct the absolute path given the current working directory for
556  // either the host filesystem or target filesystem. The distinction only
557  // matters if filesystem redirection is utilized in the simulation.
558  auto path_base = std::string();
559  if (host_filesystem) {
560  path_base = hostCwd;
561  assert(!hostCwd.empty());
562  } else {
563  path_base = tgtCwd;
564  assert(!tgtCwd.empty());
565  }
566 
567  // Add a trailing '/' if the current working directory did not have one.
568  normalize(path_base);
569 
570  // Append the filename onto the current working path.
571  auto absolute_path = path_base + filename;
572 
573  return absolute_path;
574 }
