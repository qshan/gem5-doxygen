172 void
173 LinuxAlphaSystem::setDelayLoop(ThreadContext *tc)
174 {
175  Addr addr = 0;
176  if (kernelSymtab->findAddress("loops_per_jiffy", addr)) {
177  Tick cpuFreq = tc->getCpuPtr()->frequency();
178  assert(intrFreq);
179  PortProxy &vp = tc->getVirtProxy();
180  vp.write(addr, (uint32_t)((cpuFreq / intrFreq) * 0.9988),
181  GuestByteOrder);
182  }
183 }
