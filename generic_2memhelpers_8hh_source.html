153 template <ByteOrder Order, class XC, class MemT>
154 Fault
155 writeMemAtomic(XC *xc, Trace::InstRecord *traceData, const MemT &mem,
156  Addr addr, Request::Flags flags, uint64_t *res)
157 {
158  if (traceData) {
159  traceData->setData(mem);
160  }
161  MemT host_mem = htog(mem, Order);
162  Fault fault =
163  xc->writeMem((uint8_t *)&host_mem, sizeof(MemT), addr, flags, res);
164  if (fault == NoFault && res != NULL) {
165  if (flags & Request::MEM_SWAP || flags & Request::MEM_SWAP_COND)
166  *(MemT *)res = gtoh(*(MemT *)res, Order);
167  else
168  *res = gtoh(*res, Order);
169  }
170  return fault;
171 }
